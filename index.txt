    const menuItems = [
      { id: 1, name: "Chicken Fried Rice", price: 85 },
      { id: 2, name: "Veg Fried Rice", price: 50 },
      { id: 3, name: "Veg Noodles", price: 50 },

      { id: 5, name: "Chicken Noodles", price: 70 },
      { id: 6, name: "Samosa", price: 20 },
      { id: 7, name: "Small Samosa", price: 5 },
      { id: 8, name: "Challa Punukulu", price: 20 },
      { id: 9, name: "Curry Puff", price: 15 },
      { id: 10, name: "Egg Puff", price: 20 },
      { id: 11, name: "Bread Bajji", price: 20 }
    ];

    const order = [];

    function renderMenu() {
      const menuContainer = document.getElementById('menu');
      menuItems.forEach(item => {
        const card = document.createElement('div');
        card.className = 'food-card';
        card.innerHTML = `
          <h3>${item.name}</h3>
          <p>â‚¹${item.price}</p>
          <button onclick="addToOrder(${item.id})">Add to Order</button>
        `;
        menuContainer.appendChild(card);
      });
    }

    function addToOrder(id) {
      const item = menuItems.find(i => i.id === id);
      order.push(item);
      updateOrderSummary();
    }

    function updateOrderSummary() {
      const list = document.getElementById('order-list');
      list.innerHTML = '';

      if (order.length === 0) {
        list.innerHTML = '<li>No items added.</li>';
        document.getElementById('order-total').textContent = 'Total: â‚¹0';
        return;
      }

      order.forEach(item => {
        const li = document.createElement('li');
        li.innerHTML = `${item.name} <span>â‚¹${item.price}</span>`;
        list.appendChild(li);
      });

      const total = order.reduce((sum, item) => sum + item.price, 0);
      document.getElementById('order-total').textContent = 'Total: â‚¹' + total;
    }

    function placeOrder() {
      if (order.length === 0) {
        alert("Please add items before placing an order.");
        return;
      }

      let message = "Order Placed!\n\nItems:\n";
      order.forEach(item => {
        message += `- ${item.name} (â‚¹${item.price})\n`;
      });
      message += "\nTotal: â‚¹" + order.reduce((sum, item) => sum + item.price, 0);
      alert(message);

      order.length = 0;
      updateOrderSummary();
    }

    renderMenu();
